这是一个基于三层架构设计的跨平台数据采集程序（目前支持linux win10等 基于 c++11标准）
利用sqllite做设备配置存储，使用本地json文件做离线设备数据存储
程序入口（负责启动程序等）
协议转换层（负责程序对外的协议）
设备管理层（负责管理设备信息）
设备协议层（负责采集数据）

数据表设计
协议表 ProtocolType
	id int key #协议id
	protocol string #协议类型 例如FANUC
	protocolName string #协议名字 发那科协议
	Description string #协议描述 适用于发那科数控机床io5系列
设备表 Devices
	id int key #设备id
	pid int #协议id
	protocol string #协议类型
    ip string #设备ip地址
    port int #设备端口号
字段表 CollectionField
	id int key #采集字段id
	did int #设备id
	fieldName string #字段名
	fieldAddress string #字段地址通用 可代表id 寄存器地址 其他地址等
	fieldValueType string #字段值类型 int8 int16 int32 uint16 uint32 int64 uint64 float string等
	endianness string #字节序 modbus协议使用
	description string #字段描述

	-- 协议表
CREATE TABLE IF NOT EXISTS protocol_type (
    id INTEGER PRIMARY KEY AUTOINCREMENT,        -- 协议ID 自增主键
    protocol TEXT NOT NULL UNIQUE,               -- 协议类型（如 FANUC、Modbus）唯一约束
    protocol_name TEXT NOT NULL,                 -- 协议名称（如 发那科协议）
    description TEXT,                            -- 协议描述
);
-- 索引：加速按协议类型查询
CREATE INDEX IF NOT EXISTS idx_protocol ON protocol_type(protocol);

-- 设备表
CREATE TABLE IF NOT EXISTS devices (
    id INTEGER PRIMARY KEY AUTOINCREMENT,        -- 设备ID 自增主键
    pid INTEGER NOT NULL,                        -- 关联协议表ID
    protocol TEXT NOT NULL,                      -- 协议类型（冗余字段，方便查询）
    device_name TEXT,                            -- 扩展：设备名称（建议加上）
    description TEXT,                            -- 设备描述
    status INTEGER DEFAULT 1,                    -- 设备状态 1=正常 0=禁用
    ip TEXT NOT NULL,                            -- 设备IP地址
    port INTEGER NOT NULL,                       -- 设备端口号
    create_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    update_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

);

-- 索引：加速按协议ID/协议类型查询设备
CREATE INDEX IF NOT EXISTS idx_devices_pid ON devices(pid);
CREATE INDEX IF NOT EXISTS idx_devices_protocol ON devices(protocol);


-- 采集字段表
CREATE TABLE IF NOT EXISTS collection_field (
    id INTEGER PRIMARY KEY AUTOINCREMENT,        -- 字段ID 自增主键
    did INTEGER NOT NULL,                        -- 关联设备ID
    field_name TEXT NOT NULL,                    -- 字段名
    field_address TEXT NOT NULL,                 -- 字段地址/寄存器地址
    field_value_type TEXT NOT NULL,              -- 值类型 int8/int16/float/string 等
    read_only INTEGER NOT NULL,                  --是否只读 1是 0否
    endianness TEXT,                             -- 字节序（仅Modbus使用）
    description TEXT,                            -- 字段描述
    create_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    update_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
);

-- 索引：加速按设备ID查询字段
CREATE INDEX IF NOT EXISTS idx_field_did ON collection_field(did);
